\chapter{Result: Rectangle Lower-Bound Method}

\begin{theorem}
  \label{thm:rect_lower_bound}
  \lean{DetProtocol.det_cc_lower_bound}
  \leanok
  \uses{thm:rect_partition, def:det_cc}
  To prove $\Det(g) \ge n+1$, it suffices to show every monochromatic rectangle
  partition of $g$ has more than $2^n$ parts.
\end{theorem}
